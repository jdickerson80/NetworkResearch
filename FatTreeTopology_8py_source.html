340  def port(self, src, dst):
341  '''Get port number (optional)
342  Note that the topological significance of DPIDs in FatTreeTopo enables
343  this function to be implemented statelessly.
344  @param src source switch name
345  @param dst destination switch name
346  @return tuple (src_port, dst_port):
347  src_port: port on source switch leading to the destination switch
348  dst_port: port on destination switch leading to the source switch
349  '''
350  src_layer = self.layer(src)
351  dst_layer = self.layer(dst)
352 
353  src_id = self.id_gen(name = src)
354  dst_id = self.id_gen(name = dst)
355 
356  LAYER_CORE = 0
357  LAYER_AGG = 1
358  LAYER_EDGE = 2
359  LAYER_HOST = 3
360 
361  if src_layer == LAYER_HOST and dst_layer == LAYER_EDGE:
362  src_port = 0
363  dst_port = (src_id.host - 2) * 2 + 1
364  elif src_layer == LAYER_EDGE and dst_layer == LAYER_CORE:
365  src_port = (dst_id.sw - 2) * 2
366  dst_port = src_id.pod
367  elif src_layer == LAYER_EDGE and dst_layer == LAYER_AGG:
368  src_port = (dst_id.sw - self.k / 2) * 2
369  dst_port = src_id.sw * 2 + 1
370  elif src_layer == LAYER_AGG and dst_layer == LAYER_CORE:
371  src_port = (dst_id.host - 1) * 2
372  dst_port = src_id.pod
373  elif src_layer == LAYER_CORE and dst_layer == LAYER_AGG:
374  src_port = dst_id.pod
375  dst_port = (src_id.host - 1) * 2
376  elif src_layer == LAYER_AGG and dst_layer == LAYER_EDGE:
377  src_port = dst_id.sw * 2 + 1
378  dst_port = (src_id.sw - self.k / 2) * 2
379  elif src_layer == LAYER_CORE and dst_layer == LAYER_EDGE:
380  src_port = dst_id.pod
381  dst_port = (src_id.sw - 2) * 2
382  elif src_layer == LAYER_EDGE and dst_layer == LAYER_HOST:
383  src_port = (dst_id.host - 2) * 2 + 1
384  dst_port = 0
385  else:

392  dst_port += 1
