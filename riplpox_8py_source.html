129  def _ecmp_hash(self, packet):
130  "Return an ECMP-style 5-tuple hash for TCP/IP packets, otherwise 0."
131  hash_input = [0] * 5
132  if isinstance(packet.next, ipv4):
133  ip = packet.next
134  hash_input[0] = ip.srcip.toUnsigned()
135  hash_input[1] = ip.dstip.toUnsigned()
136  hash_input[2] = ip.protocol
137  if isinstance(ip.next, tcp) or isinstance(ip.next, udp):
138  l4 = ip.next
139  hash_input[3] = l4.srcport
140  hash_input[4] = l4.dstport
141  return crc32(pack('LLHHH', *hash_input))
142  return 0
